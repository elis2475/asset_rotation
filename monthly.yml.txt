name: Monthly Asset Rotation

on:
  schedule:
    - cron: "15 6 1 * *"  # runs 06:15 UTC on day 1 each month
  workflow_dispatch:

permissions:
  contents: write

jobs:
  run:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Install deps
        run: pip install -r requirements.txt

      - name: Run pipeline
        run: python main.py

      - name: Commit outputs
        run: |
          git config user.name "github-actions"
          git config user.email "github-actions@github.com"
          git add results docs/latest.csv
          git commit -m "Monthly update" || echo "No changes to commit"
          git push
